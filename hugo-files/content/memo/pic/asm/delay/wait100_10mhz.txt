TCNT    EQU  0x20    ; 遅延発生ルーチン用の変数をこんな感じで用意してください

;==============================================================================
; 100μsec の遅延時間 ─ 10MHz (周期 0.4μsec) の時
;==============================================================================
WAIT100_10MHZ
    MOVLW  D'81'     ; 100 / 0.4 = 250
    MOVWF  TCNT      ; (250 - 6) / 3 = 81 余り 1
WAIT100_10MHZ_LOOP
    DECFSZ TCNT, F   ;
    GOTO   WAIT100_10MHZ_LOOP
    NOP              ; 余り 1 だから NOP は 1 つ
    RETURN           ;
