```asm
;==============================================================================
;
; File  : ra0_test.asm
; Author: Masatoshi OHTA
; Update: 2000-08-20
;
;==============================================================================
    LIST     P=PIC16F84
    INCLUDE  "P16F84.INC"
    __CONFIG _WDT_OFF
             ; ウォッチドッグ・タイマ OFF
             ;

;******************************************************************************
; Entry Point
;******************************************************************************
    ORG      0
    clrf     PCLATH
    goto     Start

;******************************************************************************
; Main Routine
;******************************************************************************
Start
;**** Set I/O Mode
SetIoMode
    bsf     STATUS, RP0      ; Bank1 に切り替え
    clrf    TRISA            ; PORTA をすべて出力モードに
    bcf     STATUS, RP0      ; Bank0 に戻す

;**** RA0 を ON/OFF
MainLoop
    bsf     PORTA, 0         ; RA0 を High に
    nop
    bcf     PORTA, 0         ; RA0 を Low に
    nop
    goto    MainLoop

;******************************************************************************
; Directive 'End of Program'
;******************************************************************************
    END
```
