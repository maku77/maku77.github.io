def int2str(n, base):
    if not 2 <= base <= 36:
        raise ValueError('base must be between 2 and 36')

    table = '0123456789abcdefghijklmnopqrstuvwxyz'
    buf = []
    while True:
        n, r = divmod(n, base)
        buf.append(table[r])
        if n == 0: break
    return ''.join(reversed(buf))

# Test
print(int2str(65530, 16))  #=> 'fffa'
print(int2str(64, 8))  #=> '100'
print(int2str(255, 2))  #=> '11111111'
