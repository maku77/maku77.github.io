data class Complex(val re: Int, val im: Int) : Comparable<Complex> {
    override fun compareTo(other: Complex): Int {
        return if (re == other.re) {
            im - other.im
        } else {
            re - other.re
        }
    }
}

fun main() {
    val c1 = Complex(1, 2)
    val c2 = Complex(2, 1)
    val c3 = Complex(1, 3)
    println(c1 < c2)   // => true
    println(c1 > c2)   // => false
    println(c1 <= c3)  // => true
}
